    ```tsx
    import React from 'react';

    interface ErrorBoundaryState {
      hasError: boolean;
      error: Error | null;
    }

    export class ErrorBoundary extends React.Component<
      { children: React.ReactNode },
      ErrorBoundaryState
    > {
      constructor(props: { children: React.ReactNode }) {
        super(props);
        this.state = { hasError: false, error: null };
      }

      static getDerivedStateFromError(error: Error): ErrorBoundaryState {
        return { hasError: true, error };
      }

      render() {
        if (this.state.hasError) {
          return (
            <div className="min-h-screen bg-slate-950 flex items-center justify-center p-6">
              <div className="bg-slate-900 border border-rose-900 rounded-xl p-8 max-w-md w-full text-center space-y-4">
                <p className="text-rose-400 font-semibold text-lg">Terjadi Kesalahan</p>
                <p className="text-slate-400 text-sm">{this.state.error?.message}</p>
                <button
                  onClick={() => window.location.reload()}
                  className="bg-brand-600 hover:bg-brand-700 text-white px-4 py-2 rounded-lg text-sm"
                >
                  Muat Ulang Halaman
                </button>
              </div>
            </div>
          );
        }
        return this.props.children;
      }
    }
    ```

    **A. Ganti session management dari sessionStorage ke localStorage+expiry:**

    Di `store.tsx`, tambah konstanta di atas AppProvider:
    ```ts
    const SESSION_KEY = 'raps_session';
    const SESSION_HOURS = 8;

    function saveSession(username: string) {
      const expiresAt = Date.now() + SESSION_HOURS * 60 * 60 * 1000;
      localStorage.setItem(SESSION_KEY, JSON.stringify({ username, expiresAt }));
    }

    function loadSession(): string | null {
      try {
        const raw = localStorage.getItem(SESSION_KEY);
        if (!raw) return null;
        const { username, expiresAt } = JSON.parse(raw) as { username: string; expiresAt: number };
        if (Date.now() > expiresAt) {
          localStorage.removeItem(SESSION_KEY);
          return null;
        }
        return username;
      } catch {
        return null;
      }
    }

    function clearSession() {
      localStorage.removeItem(SESSION_KEY);
    }
    ```

    Update `setup()` di useEffect â€” ganti `sessionStorage.getItem('raps_user')` dengan `loadSession()`.

    Update `login()` â€” ganti `sessionStorage.setItem('raps_user', username)` dengan `saveSession(username)`.

    Update `logout()` â€” ganti `sessionStorage.removeItem('raps_user')` dengan `clearSession()`.

    Update `updateCredentials()` â€” ganti `sessionStorage.setItem('raps_user', newU)` dengan `saveSession(newU)`.

    **B. Tambah username salt ke password hashing:**

    Ganti fungsi `hashPassword` agar menerima salt:
    ```ts
    const hashPassword = async (password: string, salt: string = ''): Promise<string> => {
      const msgBuffer = new TextEncoder().encode(password + salt);
      const hashBuffer = await crypto.subtle.digest('SHA-256', msgBuffer);
      const hashArray = Array.from(new Uint8Array(hashBuffer));
      return hashArray.map(b => b.toString(16).padStart(2, '0')).join('');
    };
    ```

    Update pemanggil `hashPassword`:
    - Di `login()`: `await hashPassword(password, username)` â€” gunakan username yang diinput sebagai salt
    - Di `updateCredentials()`: `await hashPassword(oldP, creds.username)` untuk verifikasi, `await hashPassword(newP, newU)` untuk hash baru
    - Di `resetCredentials()`: `await hashPassword('admin', 'admin')` â€” karena default user adalah 'admin'

    **PENTING:** Salt ini TIDAK mengubah hash yang sudah tersimpan di database. Setelah plan ini diterapkan, admin harus update credentials sekali melalui halaman Settings agar hash lama (tanpa salt) diganti hash baru (dengan salt). Tambahkan komentar di kode ini.
